 11 typedef struct csllnode {
 12     int val;
 13     struct csllnode *next;
 14 } *csllnode;

 17     csllnode prev;

 19     size_t index;
 20 } TupleX3;

 39 // allocate a new node
 40 csllnode new_node(int val)
 41 {
 42     csllnode node = calloc(1, sizeof(struct csllnode));
 43     if (!node) return node;
 44     node->val = val;
 45     node->next = NULL;
 46     return node;
 47 }
 48 
 49 // get node of an index
 50 csllnode csll_get(csllnode head, size_t index)
 51 {
 52     if (!head) return NULL;
 53     if (index >= head->val) {
 54         printf ("csllist: access: index out of bounds\n");
 55         return NULL;
 56     }
 57     size_t i;
 58     csllnode p = head;
 59     for (i = 0; i <= index; i++) {
 60         if (p->next == head)
 61             break;
 62         p = p->next;
 63     }
 64     return p;
 65 }

 67 // traverse csll by using a callback method
 68 bool csll_traverse(csllnode head, void(*callback)(size_t index, csllnode node))
 69 {
 70     if (!head) return false;
 71     size_t i = 0;
 72     csllnode p = head->next;
 73     while (p != head) {
 74         callback(i, p);
 75         p = p->next;
 76         i++;
 77     }
 78     return true;
 79 }

 81 void csll_print(csllnode head)
 82 {
 83     if (!head) {
 84         printf("NULL\n");
 85         return;
 86     } else if (!(head->val)) {
 87         printf("{}\n");
 88         return;
 89     }
 90     csllnode p = head->next;
 91     printf("{ ");
 92     while (p != head) {
 93         printf("%d%s", p->val, p->next != head ? ", " : " ");
 94         p = p->next;
 95     }
 96     printf("}\n");
 97 }

 99 // return a tuple of index and node where target value is located
100 TupleX3 csll_search(csllnode head, int val)
101 {
102     if (!head || !head->next) {
103         TupleX3 rslt = { NULL, NULL, 0 };
104         return rslt;
105     }
106     csllnode p = head;
107     size_t i = 0;
108     while (p->next != head) {
109         if (p->next->val == val) {
110             TupleX3 rslt = { p, p->next, i };
111             return rslt;
112         }
113         p = p->next;
114         i++;
115     }
116     TupleX3 rslt = { p, p->next, i };
117     return rslt;
118 }

120 TupleX3 csll_getmin(csllnode head)
121 {
122     if (!head || !head->next) {
123         TupleX3 tuple = { NULL, NULL, 0 };
124         return tuple;
125     }
126     csllnode p = head->next;
127     csllnode before_min = head;
128     size_t i = 0, pos = i;
129     while (p->next != head) {
130         if (p->next->val < before_min->next->val) {
131             before_min = p;
132             pos = i +1;
133         }
134         p = p->next;
135         i++;
136     }
137     TupleX3 tuple = { before_min, before_min->next, pos };
138     return tuple;
139 }

172 // insert by index, true if successful
173 bool csll_insert(csllnode head, size_t index, int val)
174 {
175     if (!head) return false;
176     if (index > head->val) {
177         printf ("csllist: insert: index out of bounds\n");
178         return false;
179     }
180     size_t i;
181     csllnode p = head;
182     for (i = 0; i < index; i++) {
183         if (p->next == head)
184             break;
185         p = p->next;
186     }
187     csllnode newnode = new_node(val);
188     if (!newnode) return false;
189     newnode->next = p->next;
190     p->next = newnode;
191     (head->val)++;
192     return true;
193 }
194 
195 // return true if insertion successful
196 bool csll_insertAftVal(csllnode head, int searchVal, int val)
197 {
198     if (!head) return false;
199     TupleX3 loc = csll_search(head, searchVal);
200     if (!loc.this) return false;
201     csllnode p = loc.this;
202     csllnode newnode = new_node(val);
203     if (!newnode) return false;
204     newnode->next = p->next;
205     p->next = newnode;
206     (head->val)++;
207     return true;
208 }

210 bool csll_delIndex(csllnode head, size_t index)
211 {
212     if (!head || head->val <= 0) return false;
213     if (index >= head->val) {
214         printf("csllist: delIndex: index out of bounds\n");
215         return false;
216     }
217     size_t i;
218     csllnode p = head;
219     for (i = 0; i < index; i++) {
220         if (p->next == head)
221             break;
222         p = p->next;
223     }
224     csllnode tmp = p->next;
225     p->next = p->next->next;
226     free(tmp);
227     (head->val)--;
228     return true;
229 }
230 
231 bool csll_delValue(csllnode head, int val)
232 {
233     if (!head || head->val <= 0) return false;
234     TupleX3 tuple = csll_search(head, val);
235     csllnode p = tuple.prev;
236     csllnode tmp = tuple.this;
237     if (!tmp) return false;
238     p->next = p->next->next;
239     free(tmp);
240     (head->val)--;
241     return true;
242 }
243 
244 bool csll_delLeft(csllnode head)
245 {
246     return csll_delIndex(head, 0);
247 }
248 
249 bool csll_delRight(csllnode head)
250 {
251     return csll_delIndex(head, head->val -1);
252 }

254 void csll_free(csllnode *head)
255 {
256     if (!*head) return;
257     csllnode p = *head;
258     while (p != *head) {
259         csllnode tmp = p->next;
260         free(p);
261         p = tmp;
262     }
263     *head = NULL;
264 }
