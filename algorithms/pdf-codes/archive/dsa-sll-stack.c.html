  7 #include <stdio.h>
  8 #include <stdlib.h>
  9 #include <stdbool.h>
 10 
 11 typedef int StackType;
 12 
 13 typedef struct stack_node {
 14     StackType val;
 15     struct stack_node *next;
 16 } *stack_node;
 17 
 18 typedef struct stack {
 19     struct stack_node *top;
 20 } *stack;
 21 
 22 stack new_stack();
 23 bool stack_push(stack st, StackType val);
 24 StackType stack_pop(stack st);
 25 StackType stack_peek(stack st);
 26 bool stack_isempty(stack st);
 27 void stack_free(stack *head);
 28 
 29 #define stack_error(e) {                     \
 30     fprintf(stderr, "stack error: %s\n", e); \
 31     abort();                                 \
 32 }
 33 
 34 #define stack_warn(e, retval) {              \
 35     fprintf(stderr, "stack warn: %s\n", e);  \
 36     return retval;                           \
 37 }
 38 
 39 // allocate a new stack
 40 stack new_stack()
 41 {
 42     stack st = calloc(1, sizeof(struct stack));
 43     if (!st) stack_error("null pointer");
 44     st->top = NULL;
 45     return st;
 46 }
 47 
 48 bool stack_push(stack st, StackType val)
 49 {
 50     if (!st) stack_error("null pointer");
 51     stack_node node = calloc(1, sizeof(struct stack_node));
 52     if (!node) stack_error("null pointer");
 53     node->val = val;
 54     node->next = NULL;
 55     if (st->top) node->next = st->top;
 56     st->top = node;
 57     return true;
 58 }
 59 
 60 StackType stack_pop(stack st)
 61 {
 62     if (!st) stack_error("null pointer");
 63     if (!st->top) stack_warn("underflow", 0);
 64     StackType tmp = st->top->val;
 65     stack_node delnode = st->top;
 66     st->top = st->top->next;
 67     free(delnode);
 68     return tmp;
 69 }
 70 
 71 StackType stack_peek(stack st)
 72 {
 73     if (!st) stack_error("null pointer");
 74     if (!st->top) stack_warn("underflow", 0);
 75     return st->top->val;
 76 }
 77 
 78 bool stack_isempty(stack st)
 79 {
 80     return !st->top;
 81 }
 82 
 83 void stack_free(stack *st)
 84 {
 85     if (!st || !*st) stack_error("null pointer");
 86     stack_node p = (*st)->top;
 87     while (p) {
 88         free(p);
 89         p = p->next;
 90     }
 91     free(*st);
 92     *st = NULL;
 93 }

119                 stack_push(st, val);
120                 break;
121             }
122             // peek
123             case 2: {
124                 printf("value at top = %d\n", stack_peek(st));
125                 break;
126             }
127             // pop
128             case 3: {
129                 printf("value on pop = %d\n", stack_pop(st));
130                 break;
131             }
132             // is empty
133             case 4: {
134                 if (stack_isempty(st))
135                     printf("stack is empty\n");
136                 else
137                     printf("stack is not empty\n");
138             }
139             default: {
140                 printf("choice invalid\n");
141             }
142         }
143     } while(ch);
144     stack_free(&st);
145     return 0;
146 }
