 11 # if !defined(MTX_TYPE) || !defined(TYPE_FORMAT)
 12 #    undef  MTX_TYPE
 13 #    undef  TYPE_FORMAT
 14 #    define MTX_TYPE     int
 15 #    define TYPE_FORMAT  "%d"
 16 # endif
 17 
 18 # define ERR_NULLPTR       (200)
 19 # define ERR_OUTOFBOUNDS   (201)
 20 # define ERR_MATINCOM      (202)
 21 # define ERR_MENUDRIVELIM  (204)
 22 
 23 # define MENUDRIVE_LIMIT  ((size_t) 10000)
 24 
 25 typedef MTX_TYPE  MatrixType;
 26 
 27 typedef struct Matrix {
 28     MatrixType *ptr;
 29     size_t rows;
 30     size_t cols;
 31 } *Matrix;
 32 
 33 void mtx_nullPtrCheck(char *fname, Matrix mtx);
 34 Matrix new_matrix(size_t rows, size_t cols);
 35 Matrix mtx_arrayToMatrix(size_t rows, size_t cols, MatrixType arr[rows][cols]);
 36 Matrix mtx_copy(Matrix mtx);
 37 void mtx_print(Matrix mtx);
 38 MatrixType mtx_get(Matrix mtx, size_t row, size_t col);
 39 void mtx_set(Matrix mtx, size_t row, size_t col, MatrixType val);
 40 Matrix mtx_add(Matrix mtx1, Matrix mtx2);
 41 Matrix mtx_scale(Matrix mtx, MatrixType scalar);
 42 Matrix mtx_multiply(Matrix mtx1, Matrix mtx2);
 43 Matrix mtx_transpose(Matrix mtx);
 44 void mtx_free(Matrix *mtx_ptr);
 45 
 46 void mtx_nullPtrCheck(char *fname, Matrix mtx)
 47 {
 48     if (!mtx || !mtx->ptr) {
 49         printf("matrix: %s: null pointer\n", fname);
 50         exit(ERR_NULLPTR);
 51     }
 52 }
 53 
 54 Matrix new_matrix(size_t rows, size_t cols)
 55 {
 56     Matrix mtx = malloc(sizeof(struct Matrix));
 57     if (!mtx) {
 58         printf("matrix: new1: null pointer\n");
 59         exit(ERR_NULLPTR);
 60     }
 61     mtx->ptr = malloc(rows * cols * sizeof(MatrixType));
 62     mtx->rows = rows;
 63     mtx->cols = cols;
 64     mtx_nullPtrCheck("new2", mtx);
 65     return mtx;
 66 }
 67 
 68 Matrix mtx_arrayToMatrix(size_t rows, size_t cols, MatrixType arr[rows][cols])
 69 {
 70     size_t i, j;
 71     Matrix mtx = new_matrix(rows, cols);
 72     for (i = 0; i < rows; i++)
 73         for (j = 0; j < cols; j++)
 74             mtx_set(mtx, i, j, arr[i][j]);
 75     return mtx;
 76 }
 77 
 78 Matrix mtx_copy(Matrix mtx)
 79 {
 80     mtx_nullPtrCheck("copy", mtx);
 81     size_t i, j;
 82     Matrix cmtx = new_matrix(mtx->rows, mtx->cols);
 83     for (i = 0; i < cmtx->rows; i++)
 84         for (j = 0; j < cmtx->cols; j++)
 85             mtx_set(cmtx, i, j, mtx_get(mtx, i, j));
 86     return cmtx;
 87 }
 88 
 89 void mtx_print(Matrix mtx)
 90 {
 91     mtx_nullPtrCheck("print", mtx);
 92     size_t i, j;
 93     for (i = 0; i < mtx->rows; i++) {
 94         for (j = 0; j < mtx->cols; j++)
 95             printf(TYPE_FORMAT "\t", mtx_get(mtx, i, j));
 96         printf("\n");
 97     }
 98 }
 99 
100 MatrixType mtx_get(Matrix mtx, size_t row, size_t col)
101 {
102     mtx_nullPtrCheck("get", mtx);
103     if (row >= mtx->rows || col >= mtx->cols) {
104         printf("matrix: get: index out of bounds: %zu, %zu\n", row, col);
105         exit(ERR_OUTOFBOUNDS);
106     }
107     return mtx->ptr[row * mtx->rows + col];
108 }
109 
110 void mtx_set(Matrix mtx, size_t row, size_t col, MatrixType val)
111 {
112     mtx_nullPtrCheck("set", mtx);
113     if (row >= mtx->rows || col >= mtx->cols) {
114         printf("matrix: set: index out of bounds: %zu, %zu\n", row, col);
115         exit(ERR_OUTOFBOUNDS);
116     }
117     mtx->ptr[row * mtx->rows + col] = val;
118 }
119 
120 Matrix mtx_add(Matrix mtx1, Matrix mtx2)
121 {
122     mtx_nullPtrCheck("add", mtx1);
123     mtx_nullPtrCheck("add", mtx2);
124     if (mtx1->rows != mtx2->rows || mtx1->cols != mtx2->cols) {
125         printf("matrix: add: incompatible matrices\n");
126         exit(ERR_MATINCOM);
127     }
128     size_t i, j;
129     Matrix rmtx = new_matrix(mtx1->rows, mtx1->cols);
130     for (i = 0; i < rmtx->rows; i++) {
131         for (j = 0; j < rmtx->cols; j++) {
132             MatrixType sum = mtx_get(mtx1, i, j) + mtx_get(mtx2, i, j);
133             mtx_set(rmtx, i, j, sum);
134         }
135     }
136     return rmtx;
137 }
138 
139 Matrix mtx_scale(Matrix mtx, MatrixType scalar)
140 {
141     mtx_nullPtrCheck("scale", mtx);
142     size_t i, j;
143     Matrix rmtx = new_matrix(mtx->rows, mtx->cols);
144     for (i = 0; i < rmtx->rows; i++) {
145         for (j = 0; j < rmtx->cols; j++) {
146             MatrixType prod = mtx_get(mtx, i, j) * scalar;
147             mtx_set(rmtx, i, j, prod);
148         }
149     }
150     return rmtx;
151 }
152 
153 Matrix mtx_multiply(Matrix mtx1, Matrix mtx2)
154 {
155     mtx_nullPtrCheck("multiply", mtx1);
156     mtx_nullPtrCheck("multiply", mtx2);
157     if (mtx1->cols != mtx2->rows) {
158         printf("matrix: multiply: incompatible matrices\n");
159         exit(ERR_MATINCOM);
160     }
161     size_t i, j, k;
162     Matrix rmtx = new_matrix(mtx1->rows, mtx2->cols);
163     for (i = 0; i < mtx1->rows; i++) {
164         for (j = 0; j < mtx2->cols; j++) {
165             MatrixType sum = 0;
166             for (k = 0; k < mtx1->cols; k++)
167                 sum += mtx_get(mtx1, i, k) * mtx_get(mtx2, k, j);
168             mtx_set(rmtx, i, j, sum);
169         }
170     }
171     return rmtx;
172 }
173 
174 Matrix mtx_transpose(Matrix mtx)
175 {
176     mtx_nullPtrCheck("transpose", mtx);
177     Matrix tmtx = new_matrix(mtx->cols, mtx->rows);
178     size_t i, j;
179     for (i = 0; i < tmtx->rows; i++)
180         for (j = 0; j < tmtx->cols; j++)
181             mtx_set(tmtx, i, j, mtx_get(mtx, j, i));
182     return tmtx;
183 }
184 
185 void mtx_free(Matrix* mtx_ptr)
186 {
187     if (mtx_ptr && *mtx_ptr && (*mtx_ptr)->ptr) {
188         free((*mtx_ptr)->ptr);
189         free(*mtx_ptr);
190         *mtx_ptr = NULL;
191     }
192 }
